\tikzmath{
  function logistic_density(\x) {
    if \x > 0 then {
      return 12 * 4 * exp(-\x) / ((1 + exp(-\x)) * (1 + exp(-\x)));
    } else {
      return 12 * 4 * exp(\x) / ((1 + exp(\x)) * (1 + exp(\x)));
    };
  };
  function logistic(\x) {
    if \x > 0 then {
      return 12 / (1 + exp(-\x));
    } else {
      return 12 * exp(\x) / (1 + exp(\x));
    };
  };
}
