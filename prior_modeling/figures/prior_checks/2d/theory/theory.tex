\tikzmath{
  function mu(\x) {
    return 6 - 0.1 * \x - 0.01 * \x * \x + 0.003 * \x * \x * \x;
  };
  function tu(\x) {
    return 7.5 - 0.125 * \x - 0.005 * \x * \x + 0.0022 * \x * \x * \x;
  };
  function tl(\x) {
    return 3 + 0.2 * \x + 0.002 * \x * \x - 0.001 * \x * \x * \x;
  };
}
