\tikzmath{
  function ribbon(\x) {
    if \x <= -1.5 then { return 0; } else {
      if \x >= 1.5 then { return 0; } else {
        return sqrt(1 + \x * \x + \x * \x * \x * \x);
      };
    };
  };
}
